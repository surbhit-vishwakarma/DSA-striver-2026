class Solution {
    public int getSecondLargest(int[] arr) {
        // code here
        
        int max = -1, secondMax = -1;
        for(int num : arr){
            if(num > max){
                secondMax = max;
                max = num;
            }
            
            if(num > secondMax && num < max)
                secondMax = num;
        }
        
        return secondMax;
    }
}
